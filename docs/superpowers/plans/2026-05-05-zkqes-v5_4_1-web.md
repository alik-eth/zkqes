```ts
import { encodeFunctionData } from 'viem';
import { ZKQES_REGISTRY_UA_V5_4_1_ABI } from '../abi/ZKQESRegistryUAV5_4_1';

export interface RegisterV5_4_1Args extends RegisterV5_4Args {
  ageCutoffDate?: number;       // YYYYMMDD; undefined or 0 = skip age
  ageProof?: AgeProofCalldata;
}

export const EMPTY_AGE_PROOF: AgeProofCalldata = {
  a: [0n, 0n],
  b: [[0n, 0n], [0n, 0n]],
  c: [0n, 0n],
  ageQualified: 0n,
  ageCutoffDate: 0n,
  nullifierCtx: 0n,
};

export function encodeRegisterV5_4_1(args: RegisterV5_4_1Args): `0x${string}` {
  return encodeFunctionData({
    abi: ZKQES_REGISTRY_UA_V5_4_1_ABI,
    functionName: 'register',
    args: [
      args.chainProof,
      args.leafProof,
      args.leafSpki,
      args.intSpki,
      args.signedAttrs,
      args.leafSig,
      args.intSig,
      args.trustMerklePath,
      args.trustMerklePathBits,
      BigInt(args.ageCutoffDate ?? 0),
      args.ageProof ?? EMPTY_AGE_PROOF,
    ],
  });
}
```

- [ ] **Step 1: Detect AA-capable wallet**

```ts
// aaWalletAdapter.ts
export interface AAWalletAdapter {
  isAACapable: boolean;
  sendUserOp: (calldata: `0x${string}`, target: `0x${string}`) => Promise<`0x${string}`>;
}

export function detectAA(connector: Connector): AAWalletAdapter {
  // ZeroDev Kernel — internal flag
  if (connector.id === 'zerodev-kernel') return {
    isAACapable: true,
    sendUserOp: (calldata, target) => sendUserOpViaKernel(connector, calldata, target),
  };
  // Coinbase Smart Wallet — declares 4337 entry points
  if (connector.id === 'coinbase' && connector.supportsEntryPoint?.('0x0000000071727De22E5E9d8BAf0edAc6f37da032'))
    return { isAACapable: true, sendUserOp: ... };
  // EOA fallback
  return { isAACapable: false, sendUserOp: () => { throw new Error('not AA-capable'); } };
}
```

- [ ] **Step 2: ZeroDev paymaster client**

```ts
// zerodevPaymaster.ts
const PAYMASTER_URL = `https://rpc.zerodev.app/api/v2/paymaster/${import.meta.env.VITE_ZERODEV_API_KEY}`;

export async function requestPaymasterSignature(userOp: UserOperation): Promise<UserOperation> {
  const response = await fetch(PAYMASTER_URL, {
    method: 'POST',
    headers: { 'content-type': 'application/json' },
    body: JSON.stringify({
      jsonrpc: '2.0',
      method: 'pm_sponsorUserOperation',
      params: [userOp, { type: 'payg' }],
      id: 1,
    }),
  });
  const data = await response.json();
  if (data.error) throw new PaymasterRejected(data.error.message);
  return { ...userOp, paymasterAndData: data.result.paymasterAndData };
}
```
