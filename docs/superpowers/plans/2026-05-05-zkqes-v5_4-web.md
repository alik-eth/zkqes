## Task 2: SDK `buildAgeWitness`

**Files:**

- [ ] **Step 1: Witness builder**

```ts
import { Buffer } from 'buffer';
import { extractDobFromDiiaSda } from '../../cert/dob-extractor-diia-ua';  // existing equivalent

export interface BuildAgeWitnessArgs {
  signedCades: Buffer;           // CAdES envelope
  bindingId: `0x${string}`;
  ageCutoffDate: number;         // YYYYMMDD
  nullifierCtxKeccak: `0x${string}`;
}

export interface BuildAgeWitnessOutput {
  witness: Record<string, string | number>;
  publicSignals: {
    ageQualified: 0 | 1;
    ageCutoffDate: number;
    nullifierCtx: string;
  };
}

export async function buildAgeWitness(args: BuildAgeWitnessArgs): Promise<BuildAgeWitnessOutput> {
  // 1. Parse + extract dobYmd
  const { dobYmd, signedAttrsBytes, sdaFrameOffset } = extractDobFromDiiaSda(args.signedCades);

  // 2. Compute ageQualified off-circuit (matches circuit logic)
  const ageQualified: 0 | 1 = dobYmd <= args.ageCutoffDate ? 1 : 0;

  // 3. Assemble witness
  const witness = {
    signedAttrsBytes: Array.from(signedAttrsBytes),
    sdaFrameOffset,
    ageCutoffDateIn: args.ageCutoffDate,
    nullifierCtxInput: BigInt(args.nullifierCtxKeccak).toString(),
    nullifierSecret: '1',  // V5.1 carry-through; sourced from session per existing flow
  };

  return {
    witness,
    publicSignals: {
      ageQualified,
      ageCutoffDate: args.ageCutoffDate,
      nullifierCtx: BigInt(args.nullifierCtxKeccak).toString(),
    },
  };
}
```

- [ ] **Step 2: AgeProof ABI types**

`packages/sdk/src/cert/age-proof.ts`:
```ts
export interface AgeProofCalldata {
  a: [bigint, bigint];
  b: [[bigint, bigint], [bigint, bigint]];
  c: [bigint, bigint];
  ageQualified: bigint;
  ageCutoffDate: bigint;
  nullifierCtx: bigint;
}

export function packAgeProof(snarkjsProof: SnarkjsProof, publicSignals: BigNumberish[]): AgeProofCalldata {
  return {
    a: [BigInt(snarkjsProof.pi_a[0]), BigInt(snarkjsProof.pi_a[1])],
    b: /* same swap as V5.2 */,
    c: [BigInt(snarkjsProof.pi_c[0]), BigInt(snarkjsProof.pi_c[1])],
    ageQualified:  BigInt(publicSignals[0]),
    ageCutoffDate: BigInt(publicSignals[1]),
    nullifierCtx:  BigInt(publicSignals[2]),
  };
}
```
